El constructor

En programación orientada a objetos, un constructor es un método especial de una clase que se llama automáticamente cuando se crea un objeto de esa clase. Su función principal es inicializar los atributos del objeto.


Python

    # Definición de una clase en Python con un constructor
    class Persona:
        # Constructor con atributos
        def __init__(self, nombre, edad):
            self.nombre = nombre
            self.edad = edad
    
    # Crear un objeto (instancia) de la clase Persona usando el constructor
    nueva_persona = Persona(nombre="Juan", edad=25)
    
    # Acceder a los atributos del objeto
    print(nueva_persona.nombre)
    print(nueva_persona.edad)

Java

    // Definición de una clase en Java con un constructor
    public class Persona {
        // Atributos
        String nombre;
        int edad;
    
        // Constructor con parámetros
        public Persona(String nombre, int edad) {
            this.nombre = nombre;
            this.edad = edad;
        }
    
        // Método principal (main) para probar la clase
        public static void main(String[] args) {
            // Crear un objeto (instancia) de la clase Persona usando el constructor
            Persona nuevaPersona = new Persona("Juan", 25);
    
            // Acceder a los atributos del objeto
            System.out.println(nuevaPersona.nombre);
            System.out.println(nuevaPersona.edad);
        }
    }
