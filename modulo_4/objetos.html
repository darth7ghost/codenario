Objetos

En programación orientada a objetos, un objeto es una instancia o ejemplar de una clase. Una clase es como un plano o plantilla que define las propiedades (atributos) y acciones (métodos) que los objetos creados a partir de ella tendrán.


Python

    # Definición de una clase en Python
    class Coche:
        # Constructor con atributos
        def __init__(self, marca, modelo):
            self.marca = marca
            self.modelo = modelo
            self.velocidad = 0
    
        # Método para acelerar el coche
        def acelerar(self, incremento):
            self.velocidad += incremento
            print(f"El coche aceleró a {self.velocidad} km/h")
    
    # Crear un objeto (instancia) de la clase Coche
    mi_coche = Coche(marca="Toyota", modelo="Corolla")
    
    # Acceder a los atributos y llamar a un método
    print(mi_coche.marca)
    mi_coche.acelerar(20)

Java

    // Definición de una clase en Java
    public class Coche {
        // Atributos
        String marca;
        String modelo;
        int velocidad;
    
        // Constructor con parámetros
        public Coche(String marca, String modelo) {
            this.marca = marca;
            this.modelo = modelo;
            this.velocidad = 0;
        }
    
        // Método para acelerar el coche
        public void acelerar(int incremento) {
            this.velocidad += incremento;
            System.out.println("El coche aceleró a " + this.velocidad + " km/h");
        }
    
        // Método principal (main) para probar la clase
        public static void main(String[] args) {
            // Crear un objeto (instancia) de la clase Coche
            Coche miCoche = new Coche("Toyota", "Corolla");
    
            // Acceder a los atributos y llamar a un método
            System.out.println(miCoche.marca);
            miCoche.acelerar(20);
        }
    }
